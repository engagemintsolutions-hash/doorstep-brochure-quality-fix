+/**
+ * Extract property data from brochure editor state
+ */
+function getPropertyDataFromBrochure() {
+    // Try to get data from window.brochureData or editor state
+    const editorTextarea = document.getElementById('editor');
+    const propertyNameInput = document.getElementById('propertyName');
+    const addressInput = document.getElementById('address');
+    const priceInput = document.getElementById('price');
+    const bedroomsInput = document.getElementById('bedrooms');
+    const bathroomsInput = document.getElementById('bathrooms');
+    const propertyTypeInput = document.getElementById('propertyType');
+
+    // Extract key features from editor text (look for bullet points)
+    const editorText = editorTextarea?.value || '';
+    const keyFeatures = [];
+    const bulletMatches = editorText.match(/[‚Ä¢\-\*]\s*([^\n]+)/g);
+    if (bulletMatches) {
+        bulletMatches.forEach(match => {
+            const cleaned = match.replace(/^[‚Ä¢\-\*]\s*/, '').trim();
+            if (cleaned.length > 0) keyFeatures.push(cleaned);
+        });
+    }
+
+    return {
+        name: propertyNameInput?.value || 'Luxury Property',
+        address: addressInput?.value || 'Prime Location',
+        price: priceInput?.value || '',
+        bedrooms: bedroomsInput?.value ? parseInt(bedroomsInput.value) : null,
+        bathrooms: bathroomsInput?.value ? parseInt(bathroomsInput.value) : null,
+        property_type: propertyTypeInput?.value || '',
+        key_features: keyFeatures.length > 0 ? keyFeatures : null,
+        description: editorText
+    };
